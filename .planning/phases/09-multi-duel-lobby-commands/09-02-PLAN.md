```rust
#[derive(Subcommand)]
enum Commands {
    Wallet(commands::wallet::WalletArgs),
    Register { nickname: String },
    // ... other variants
}

// Wallet dispatch (no-wallet path):
Commands::Wallet(_) => {
    if let Commands::Wallet(args) = cli.command {
        return commands::wallet::execute(args, agent).await;
    }
}

// Queue dispatch (wallet-required path):
Commands::Queue { bet_tier, timeout, games } => {
    let client = HttpClient::new(&backend_url, signer.clone(), address, &private_key_hex)?;
    commands::queue::execute(&client, bet_tier, timeout, &address, &signer, &rpc_url, fmt, games).await
}
```

<!-- Shell auto-wiring note -->
The shell works by calling `Cli::try_parse_from(&full_args)` then `run(cli)`.
Adding a variant to Commands enum automatically makes it available in shell.
No explicit shell.rs changes needed.
